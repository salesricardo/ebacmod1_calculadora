#!/bin/bash
# A instrução acima indica ao sistema operacional que o script deve ser interpretado usando o Bash.
# Ela é chamada de Shebang.

# Cria diretório em home para prática de exercício.
echo "Criando diretório para prática de exercício em /home/ebac_alunos/ricardods..."
sudo mkdir /home/ebac_alunos/ricardods

# Move Terminal para local criado.
echo "Movendo Terminal para local criado..."
cd /home/ebac_alunos/ricardods

# O editor nano é aberto para colar o código Python da calculadora.
# O código foi feito em outro exercício de um módulo anterior.
# Não foi apresentado como automatizar essa tarefa, então é
# assumido que isso é feito manualmente pelo usuário dentro do nano.
# Ao abrir o nano basta utilizar o botão direito do mouse para colar
# o conteúdo do código Python copiado. CTRL + X é a opção para
# salvar o arquivo.
echo "Abrindo editor para criação de calculadora em Python..."
sudo nano calculadora.py

# As permissões da calculadora são alteradas.
# 7 = Read (4) + Write (2) + Execute (1) = 7 para usuário.
# 4 = Read (4) para grupo.
# 4 = Read (4) para outros.
echo "Atribuindo permissões 744 para programa de calculadora..."
sudo chmod 744 calculadora.py

# Garante que python esteja instalado para execução.
# Sem isso Python pode não estar instalado, levando a um erro.
echo "Garantindo ambiente Python para execução de programa..."
sudo apt install python3

# Executa programa de calculadora.
echo "Executando programa de calculadora criado (calculadora.py)..."
python3 calculadora.py

# Finaliza script.
echo "Fim de execução de Script."
